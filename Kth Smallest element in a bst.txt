/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int kthSmallest(TreeNode* root, int k) {
        vector<int> result;
        
        if(!root ){
            return 0;
        }
        
        if(root->left == NULL && root->right == 0){
            return root->val;
        }
        
        
        
        iteratorv(root,result);
        
        return result[k-1];
    }
    
    void iteratorv(TreeNode * root, vector<int> &result){
        if(root == NULL){
            return;
        }
        else{
            iteratorv(root->left,result);
            
            result.push_back(root->val);
            
            iteratorv(root->right, result);
            
            
        }
    }
    
    
    
    
    
    
};
