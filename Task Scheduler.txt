public class Solution {
    public int leastInterval(char[] tasks, int n) {
        char[] arr = new char[26];
        
        for(int i = 0; i < tasks.length; i++){
            arr[tasks[i]-'A']++;
        }
        
        Arrays.sort(arr);
        
        int i = 0;
         
        for( i = 25; i >= 0; i--){
            if(arr[25] != arr[i]){
                break;
            }
        }

        
        return Math.max((n+1)*(arr[25]-1) +  25 - i, tasks.length);
        
    }
}
