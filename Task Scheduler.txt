public class Solution {
    public int leastInterval(char[] tasks, int n) {
       int[] cnt = new int[26];
        
        for(int i = 0; i < tasks.length; i++){
            cnt[tasks[i] - 'A']++;
        }
        
        Arrays.sort(cnt);
        
        int i = 25;
       
        while(i >= 0 && cnt[i] == cnt[25]){
            i--;
        }
        
        System.out.println(cnt[25] + "   " + i);
        
        return Math.max( ( cnt[25] - 1 )* (n+1) + 25 - i, tasks.length);
    }
}
