class Solution {
public:
    int numSquares(int n) {
        vector<int> square (n+1,INT_MAX);
        
        for (int i = 1; i*i <= n; i++){
            square[i*i] = 1;
        }
        
        for(int i = 1; i <= n; i++){
            for(int j = 1; i+ j*j <= n; j++){
                square[ i+ j*j] = min(square[i]+1, square[i + j*j]);
            }
        }
        
        return square[n];
        
    }
};
