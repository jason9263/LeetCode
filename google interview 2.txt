#include <iostream>
#include <string>
#include <vector>
using namespace std;
long factorial(long n){
    long fact = 1;
    for (long i = 1; i<= n; ++i)
        fact *= i;
    return fact;
}

void DFS(const string &s1, int m, int i,
         const string&s2, int n, int j,
         string path,vector<string> &ret)
{
    if(i==m &&j==n)  {
       ret.push_back(path);
        return;
    }
    if(i < m)  DFS(s1, m, i+1, s2, n, j, path+s1, ret);
    if(j < n)  DFS(s1, m, i, s2, n, j+1, path+s2[j], ret);
}

vector<string> combineTowStrings(const string &s1,const string &s2) {
    int m =s1.length(), n = s2.length();
    long count =factorial(m+n)/factorial(n)/factorial(m);
    cout <<"count:" << count << endl;
   vector<string> ret;
    DFS(s1, m, 0, s2,n, 0, "", ret);
    return ret;
}

int main() {
   vector<string> ret = combineTowStrings("AB","CDF");
    cout <<ret.size() << endl;
    for(string &s: ret)  cout << s << endl;
    return 0;

#include <iostream>
#include <string>

#include <algorithm>

bool pairCompare(const pair<int, int>& a, constpair<int, int>& b) {
    return a.first< b.first;
}
int reGroup(vector<pair<int, int>> in,vector<int> counts,
           vector<vector<pair<int, int>>> &ret)
{
    int n = in.size(),m = counts.size();
    sort(in.begin(),in.end(), pairCompare);
    ret.resize(m);
    for(int i=1; i< m; ++i){
        counts +=counts[i-1];
    }
    int j = 0;
    for(int i=0;i<n; ++i){
        int key =in.first;
        while(key>= counts[j]) {
            ++j;
            if(j >= m) {
                cout<< "Illegal input." << endl;
                return-1;
            }
        }
        cout <<key << "\tgroup=" << j << endl;
        if(j >0)  in.first -= counts[j-1];
       ret[j].push_back(in);
    }
    return 0;
}

#include <vector>
using namespace std;

void printLargerSmaller(const vector<int> &v) {
    int n = v.size();
    vector<int>leftMax(n, INT_MIN);
    for(int i=1;i<n; ++i)     leftMax  = max(leftMax[i-1], v[i-1]);
    int rightMin =INT_MAX;
    for(int i=n-1;i>=0; i--)  {
        if(i<n-1)rightMin = min(rightMin, v[i+1]);
        if(v >leftMax && v < rightMin) cout << v << " ";
    }
}

-google 1point3acres
楼主工作的时候比较懒散，算法什么的早就还个老师了，leetcode最近的一次提交也是1年多前。怎么办，送上门来的肥肉不啃感觉不开心，于是我直接问对方，面试都要准备些什么啊，对方说一般准备一下各种常规算法，系统设计之类，并且跟我强调一般成功的应聘者大多数都准备2个月甚至半年，你准备好了就给你安排电话面试，巴拉巴拉介绍电话面试流程。我说好吧，给我2星期时间准备一下再答复。

于是开始在leetcode上做些热身运动，同时在网上找垫脚石，于是大量投简历。那2-3个星期，接到了很多其他公司的电话面试，都跪了，你没看错，都跪了。楼主那个心酸啊，这狗狗的面试肯定凶多吉少。于是楼主并没有主动联系狗家的人，而是继续自暴自弃的刷题，心想要是对方忘记这件事就算了，我继续在原公司得过且过，反正我不是狗家的料。

1个月后，狗家的人来邮件了，问你准备得怎么啊，语气特别亲切。我心想，跪就跪吧，总不能得罪狗家的人，人家都主动两次找上门了。于是硬着头皮说，来吧。
