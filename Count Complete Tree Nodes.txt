/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int countNodes(TreeNode* root) {
        
        if(root == NULL){
            return 0;
        }
        
        
        int left = LeftHeight(root->left);
        int right = RightHeight(root->right);
        
        if(left != right){
            
            return 1 + countNodes(root->left) + countNodes(root->right);
            
            
            
        }else{
            return 1 + 2*(pow(2,left)-1);
        }
        
        
    }
    
    int LeftHeight(TreeNode * root){
        
        
        if(root == NULL){
            return 0;
        }else{
            int height = 0;
            while(root != NULL){
                height++;
                root = root->left;
            }
            
            return height;
        }
    }
    
    
    
    int RightHeight(TreeNode *root){
        if(root == NULL){
            return 0;
        }else{
            int height = 0;
            while(root != NULL){
                height++;
                root = root->right;
                
            }
            
            return height;
        }
    }
    

    
    
    
};
