public class Solution {
    public String reverseVowels(String s) {
        String vow = "aeiouAEIOU";
        int first = 0;
        int last = s.length()-1;
        
        char[] vowarray = s.toCharArray();
        
        
        while(first < last){
            
            while(first < last && vow.indexOf(vowarray[first]) == -1){
                first++;
            }
            
            
            while(first < last && vow.indexOf(vowarray[last]) == -1){
                last--;
            }
            
            
            char tmp = vowarray[first];
            vowarray[first] = vowarray[last];
            vowarray[last] = tmp;
            
            first++;
            last--;
            
            
            
        }
        
        return String.valueOf(vowarray);
        
    }
}
