public class Solution {
    public int maxProfit(int[] prices) {
        int max = 0;
        if(prices.length == 0){
            return max;
        }
        int sofarmin = prices[0];

        for(int i = 0; i < prices.length; i++){
            if(prices[i] > sofarmin){
                max = Math.max(max, prices[i] - sofarmin);
            }else{
                sofarmin = prices[i];
            }
        }

        return max;
    }
}
