class Solution { public: int hIndex(vector<int>& citations) { if(citations.size() == 0){ return 0; } if(citations.size() == 1){ if(citations[0] > 0){ return 1; }else{ return 0; } } sort(citations.begin(),citations.end()); int sizec = citations.size(); int hmax = 0; for(int i = 0 ; i < sizec; i++){ if(sizec - i >= citations[i] || citations[i] >= sizec - i){ hmax = max(min(sizec - i , citations[i]) , hmax); } } return hmax; } };
