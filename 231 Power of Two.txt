public class Solution {
    public boolean isPowerOfTwo(int n) {
        if(n <= 0){
            return false;
        }else{
            while( n > 1){
                if(n % 2 == 0){
                    n = n /2;
                }else{
                    return false;
                }
            }
            
            return true;
        }
    }
}
