public class Solution {
    public boolean wordBreak(String s, Set<String> wordDict) {
        boolean [] f = new boolean[s.length() +1];
        f[0] = true;
        
        
        for(int i = 1; i < s.length()+1; i++){
            for(int j = 0 ; j < i ;j++){
                if(f[j] && wordDict.contains(s.substring(j,i))){
                    f[i] = true;
                    break;
                }
            }
        }
        
        return f[s.length()];
        
    }
}
