public class Solution {
    public String reverseWords(String s) {
        String res = "";
        
        if(s ==  null || s.length() == 0){
            return res;
        }
        
        String[] substr = s.split(" ");
        StringBuffer sbuf = new StringBuffer();
        
        for(int i = 0; i < substr.length- 1; i++){
            StringBuffer tmps = new StringBuffer(substr[i]);
            tmps = tmps.reverse();
            tmps.append(' ');
            
            sbuf.append(tmps);
        }
        StringBuffer tmps = new StringBuffer(substr[substr.length-1]);
        tmps = tmps.reverse();
        
        sbuf.append(tmps);
        
        return sbuf.toString();
        
        
    }
}
