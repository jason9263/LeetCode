public class Solution {
    public List<Integer> majorityElement(int[] nums) {
        int cntm = 0, cntn = 0, m = 0, n = 0;
        
        for(int i = 0; i < nums.length; i++){
            int tmp = nums[i];
            if(m == tmp){
                cntm++;
            }else if(n == tmp){
                cntn++;
            }else if(cntm == 0){
                cntm = 1;
                m = tmp;
            }else if(cntn == 0){
                cntn = 1;
                n = tmp;
            }else{
                cntm--;
                cntn--;
            }
        }
        
        cntm = 0;
        cntn = 0;
        
        for(int i = 0; i < nums.length; i++){
            int tmp = nums[i];
            if(tmp == m){
               cntm++; 
            }else if(tmp == n){
                cntn++;
            }
        }
        int len = nums.length;
        
        List<Integer> res = new ArrayList<Integer>();
        if(cntm > len/3){
            res.add(m);
        }
        
        if(cntn > len/3){
            res.add(n);
        }
        
        return res;
    }
}
