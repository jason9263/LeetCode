public class Solution {
    public int maxArea(int[] height) {
        
        int res = 0;
        
        
        if(height == null || height.length == 1 ){
            return res;
        }
        
        
        int left = 0;
        int right = height.length-1;
        
        
        while(left < right){
            
            res = Math.max(res, Math.min(height[left], height[right])*(right - left));
            
            if(height[left] > height[right]){
                right--;
            }else{
                left++;
            }
            
        }
        
        return res;

    }
}
