/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == NULL){
            return NULL;
        }
        ListNode * pNewHead = NULL;
        ListNode * pre = NULL;
        ListNode * cur = head;
        while(cur != NULL){
            
            ListNode *temp = cur->next;
            if(temp == NULL){
                pNewHead = cur;
                
            }
            
            cur->next = pre;
            pre = cur;
            cur = temp;
            
            
        }
        
        return pNewHead;
        
        
    }
};
