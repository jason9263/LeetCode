/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* insertionSortList(ListNode* head) {
        if(!head || !head->next){
            return head;
        }
        
        ListNode *fakehead = new ListNode(INT_MIN);

        
        
        ListNode *cur = head;
        
        while( cur != NULL){
            
            ListNode * nextcur = cur->next;
            cur->next = NULL;
            
            ListNode *pre = fakehead;
            ListNode *prepre;
            
            while(pre!= NULL && pre->val <= cur->val){
                prepre = pre;
                pre  = pre->next;
            }
                prepre->next = cur;
                cur ->next = pre;
                
            cur = nextcur;  
                
        }
        
        return fakehead->next;
        
        
    }
};
