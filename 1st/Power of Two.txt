class Solution {
public:
    bool isPowerOfTwo(int n) {
        if(n <= 0){
            return false;
        }
        
        while(true){
            
            if(n%2 == 0 && n/2 != 1){
                n = n/2;
            }else if( n == 1 || n == 2){
                return true;
                
                }else
                
                {
                return false;
            }
            
        }
        
    }
};
