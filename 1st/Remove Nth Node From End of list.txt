/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        
        if(!head){
            return NULL;
        }
        
        if(!head->next && n == 1){
            return NULL;
        }
        
        
        
        
        
        
        
     
        
        ListNode *virhead ;
        virhead->val = 0;
        virhead ->next =head;
        
        
        ListNode *first = virhead;
        ListNode *second = virhead;
        
        int index = 0;
        
        while(index <= n){
            first = first->next;
            index++;
        }
        
        while(first != NULL){
            first = first->next;
            second = second->next;
        }
        
        ListNode *temp = second->next;
        second ->next = temp->next;
        
        return virhead->next;
        
    }
};
