class Solution {
public:
    int countDigitOne(int n) {
        int res = 0;
        int a = 1;
        int b = 1;
        
        
        while(n > 0){
            res = res + (n + 8) / 10 * a + (n % 10 == 1)*b;
            b = b + n %10 * a;
            
            a = a * 10;
            n = n / 10;
        }
        
        return res;
        
        
    }
};
