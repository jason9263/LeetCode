public class Solution {
    public int scheduleCourse(int[][] courses) {
        Arrays.sort(courses,(a,b)->a[1] - b[1]);
        
        PriorityQueue<Integer> p = new PriorityQueue<Integer>((a,b)-> b-a);
        int cur = 0;
        
        for(int[] c : courses){
            cur += c[0];
            p.add(c[0]);
            
            if(cur > c[1]){
                cur -= p.poll();
            }
            
        }
        
        return p.size();
        
    }
}
