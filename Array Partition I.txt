public class Solution {
    public int arrayPairSum(int[] nums) {
        int res = 0;
        if(nums == null || nums.length == 0){
            return res;
        }
        
        Arrays.sort(nums);
        int n = nums.length/2;
        
        for(int i = 0; i < n; i++){
            res += nums[i*2];
        }
        
        return res;
        
    }
}
