class Solution {
public:
    int maxArea(vector<int>& height) {
        
        if(height.size() == 0 || height.size() == 1){
            return 0;
        }else{
            
        int maxvolume = 0;
        int Lindex = 0;
        int Rindex = height.size()-1;
        
        
        while(Lindex < Rindex){
            if(height[Lindex] < height[Rindex]){
                
                maxvolume = max(maxvolume, min(height[Lindex],height[Rindex])*(Rindex - Lindex));
                Lindex++;
                
            }else if(height[Lindex] > height[Rindex]){
                
                maxvolume = max(maxvolume, min(height[Lindex],height[Rindex])* (Rindex - Lindex));
                Rindex--;
                
            }else{
                
                maxvolume = max(maxvolume, min(height[Lindex],height[Rindex])* (Rindex - Lindex));
                Rindex--;
                
            }
            
            
        }
        
        return maxvolume;
            
        }
        
    }
};
