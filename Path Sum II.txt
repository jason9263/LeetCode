/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
public class Solution {
    public List<List<Integer>> pathSum(TreeNode root, int sum) {
        List<List<Integer>> res = new LinkedList<List<Integer>>();
        List<Integer> tmpres = new LinkedList<Integer>();    
        
        helper(res,tmpres,root,sum);
        
        return res;
    }
    
    
    public void helper(List<List<Integer>> res, List<Integer> tmpres, TreeNode root, int sum){
        
        if(root == null){
            return;
        }
        
        tmpres.add(root.val);
        
        if(root.left == null && root.right == null && root.val == sum){
            res.add(new LinkedList<Integer>(tmpres));
        }
        
        helper(res,tmpres,root.left,sum-root.val);
        helper(res,tmpres,root.right,sum-root.val);
        
        tmpres.remove(tmpres.size()-1);
        
    }
    
    
    
}
