public class Solution {
    public int searchInsert(int[] nums, int target) {
        if(nums == null || nums.length == 0){
            return 0;
        }
        
        if(nums.length == 1){
            return target > nums[0] ? 1:0;
        }
        
        if(target > nums[nums.length-1]){
            return nums.length;
        }
        
        if(target < nums[0]){
            return 0;
        }
        
        
        int left = 0;
        int right = nums.length -1;
        int mid = -1;
        
        while(left <= right){
            mid = (left +right)/2;
            if(nums[mid] == target){
                return mid;
            }else if(nums[mid] < target){
                left = mid+1;
            }else if(nums[mid] > target){
                right = mid -1;
            }
        }
        
        if(nums[mid] < target){
            return mid+1;
        }else{
            return mid;
        }
        
        
    }
}
