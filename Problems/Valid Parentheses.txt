public class Solution {
    public boolean isValid(String s) {
        
       Stack<Character> js = new Stack<Character>();
       
       for(int i = 0 ; i < s.length(); i++){
            char temp = s.charAt(i);
           if(temp != '(' && temp != ')' && temp != '{' && temp != '}' && temp != '[' && temp != ']'){
               return false;
           }else{
               if(temp == ')'){
                   if(js.size() != 0 && js.peek() == '('){
                       js.pop();
                   }else{
                       return false;
                   }

               }
               
               if(temp == '}'){
                   if(js.size() != 0 && js.peek() == '{'){
                       js.pop();
                   }else {
                       return false;
                   }
               }
               
               if(temp == ']'){
                   if(js.size() != 0 && js.peek() == '['){
                       js.pop();
                   }else{
                       return false;
                   }
               }
               
               
               if(temp == '(' || temp == '{' || temp == '['){
                   js.push(temp);
               }
               
               
           }
       }
       
       if(js.size() > 0){
           return false;
       }
       
       return true;
       
        
    }
}
