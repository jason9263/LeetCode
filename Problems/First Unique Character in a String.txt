public class Solution {
    public int firstUniqChar(String s) {
        int[] fre = new int[256];
        
        for(int i = 0; i < s.length(); i++){
            fre[s.charAt(i)]++;
        }
        
        for(int i = 0; i < s.length(); i++){
            if(fre[s.charAt(i)] == 1){
                return i;
            }
        }
        
        
        return -1;
    }
}
