public class Solution {
    public int minMoves(int[] nums) {
        if(nums.length == 0){
            return 0;
        }
        
        int minv = Integer.MAX_VALUE;
        
        for(int temp : nums){
            minv = Math.min(minv,temp);
        }
        
        int res = 0;
        
        for(int temp : nums){
            res += temp - minv;    
        }
        
        
        return res;
        
    }
}
