public class Solution {
    public boolean checkPerfectNumber(int num) {
        if(num == 1){
            return false;
        }
        
        int sum = 0;
        
        for(int i = 2; i <= Math.sqrt(num); i++){
            if(num % i == 0){
                System.out.println(i +"   " + num/i);
                
                sum += (i + num/i);
            }
        }
        
        sum++;
        
        return sum == num;
        
    }
}
