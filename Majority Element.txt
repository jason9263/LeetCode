public class Solution {
    public int majorityElement(int[] nums) {
        int count = 1;
        int tmp = nums[0];
        
        for(int i = 1; i < nums.length; i++){
            if(nums[i] == tmp){
                count++;
            }else{
                count--;
                if(count == 0){
                    tmp = nums[i];
                    count = 1;
                }
            }
        }
        
        
        return tmp;
        
    }
}
