public class Solution {
    public void moveZeroes(int[] nums) {
        int index = 0;
        
        if(nums == null || nums.length == 0){
            return ;
        }
        
        for(int tmp : nums){
            if(tmp != 0){
                nums[index++] = tmp;
            }
        }
        
        while(index < nums.length){
            nums[index++] = 0;
        }
    }
}
