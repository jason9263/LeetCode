public class Solution {
    public int[] plusOne(int[] digits) {
        int len = digits.length;
        
        for(int j = len-1; j >= 0; j--){
            if(digits[j] < 9){
                digits[j]++;
                return digits;
            }
            digits[j]=0;
        }
        
        
        int[] res = new int[len+1];
        res[0] = 1;
        
        return res;
        
    }
}
