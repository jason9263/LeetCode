public class Solution {
    public int hIndex(int[] citations) {
        int h = 0;
        if(citations == null || citations.length == 0){
            return h;
        }
        
        Arrays.sort(citations);
        int len = citations.length;
        
        for(int i = 0; i < len; i++){
            h = Math.max( h, Math.min(citations[i], len - i) );
        }
        
        return h;
    }
}
