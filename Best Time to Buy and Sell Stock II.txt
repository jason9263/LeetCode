public class Solution {
    public int maxProfit(int[] prices) {
        
        int res = 0;
        
        if(prices == null || prices.length == 0 || prices.length == 1){
            return res;
        }
        
        
        
        for(int i = 0; i < prices.length-1; i++){
            res += Math.max(0, prices[i+1] - prices[i]);
        }
        
        return res;
    }
}
