class Solution {
public:
    bool isAnagram(string s, string t) {

        
        if(s.length() == 0 && t.length() == 0){
            return true;
        }else if(s.length() != t.length()){
            return false;
        }
        
        sort(s.begin(), s.end());
        sort(t.begin(), t.end());
        
        return  s == t? true:false;
    }
};
