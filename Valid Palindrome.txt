public class Solution {
    public boolean isPalindrome(String s) {
        s = s.toLowerCase();
        
        StringBuilder sb = new StringBuilder("");
        
        for(int i = 0; i < s.length(); i++){
            char tmpc = s.charAt(i);
            if( (tmpc>= '0' && tmpc <= '9') || (tmpc>='a' && tmpc<='z') ){
                sb.append(tmpc);
            }
        }
        
        String tmps = sb.toString();
        
        int left = 0;
        int right = tmps.length()-1;
        
        while(left < right){
            if(tmps.charAt(left) != tmps.charAt(right)){
                return false;
            }else{
                left++;
                right--;
            }
        }        
        return true;
    }
}
