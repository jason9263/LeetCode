public class Solution {
    public int reverse(int x) {
        
        long result = 0;
        
        while(x != 0){
            
            int tail = x%10;
            result = result*10 + tail;
            
            if(Math.abs(result) > Integer.MAX_VALUE){
                return 0;
            }
            
            
            x = (x - tail)/10;
            
        }
        
        return (int)result;
        
        
    }
}
