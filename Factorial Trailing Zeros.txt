class Solution {
public:
    int trailingZeroes(int n) {
               if(n < 5){
            return 0;
        }else{
            int power = log(n)/log(5);
            int sum = 0;
            for(int i = 1;i <= power; i++){
                sum = sum + n/pow(5,i);
            }
            
            return sum;
        } 
    }
};
