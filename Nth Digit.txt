public class Solution {
    public int findNthDigit(int n) {
        int k =  1;
        long base = 1;
        
        while(n > base * 9*k ){
            n -= base*9*k;
            k++;
            base = base*10;
        }
        
        long value = base + (n-1)/k;
        
        
        return (Long.toString(value).charAt( (n-1 )%k) - '0');
        
        
    }
}
