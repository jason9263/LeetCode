public class Solution {
    public int findKthLargest(int[] nums, int k) {
        
        Queue<Integer> q = new PriorityQueue<Integer>();
        
        for(int tmp : nums){
            
            q.offer(tmp);
            
            if(q.size() > k){
                q.poll();
            }
            
        }
        
        
        
        return q.peek();
        
        
    }
}
