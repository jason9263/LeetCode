public class RandomizedSet {
    
    public HashMap<Integer, Integer> jhash = new HashMap<Integer, Integer>();
    public ArrayList<Integer> jarr = new ArrayList<Integer>();
    public Random jr = new Random();
    
    /** Initialize your data structure here. */
    public RandomizedSet() {

    }
    
    /** Inserts a value to the set. Returns true if the set did not already contain the specified element. */
    public boolean insert(int val) {
        if(jhash.containsKey(val)){
            return false;
        }else{
            int len = jarr.size();
            
            jarr.add(val);
            jhash.put(val, len);
            
            return true;
        }
    }
    
    /** Removes a value from the set. Returns true if the set contained the specified element. */
    public boolean remove(int val) {
        if(jhash.containsKey(val)){
            
            
            int pos = jhash.get(val);
            int len = jarr.size();
            int tmpval = -1;
            
            if(pos < len-1){
                tmpval = jarr.get(len-1);
                jhash.put(tmpval,pos);
                jarr.set(pos,tmpval);
            }
            
            jarr.remove(jarr.size() - 1);
            
            jhash.remove(val);
            
            return true;
            
        }else{
            
            return false;
          
        }
    }
    
    /** Get a random element from the set. */
    public int getRandom() {


        
        if(jarr.size() > 0){
            
            int tmp = jr.nextInt(jarr.size());
            
            System.out.println(tmp + "  " + jarr.size());
            
            return jarr.get(tmp);
        }
        
        return 0;

    }
}
