public class Solution {
    public int singleNumber(int[] nums) {
        int res = 0;
        
        if(nums == null || nums.length == 0){
            return res;
        }
        
        res = nums[0];
        
        for(int i = 1; i < nums.length; i++){
            res ^= nums[i];
        }
        
        return res;
        
    }
}
