/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isValidBST(TreeNode* root) {
        return isV(root,LONG_MIN,LONG_MAX);
    }
    
    bool isV(TreeNode * root, long min, long max){
        if(!root){
            return true;
        }
        
        if(root->val > min && root->val < max && isV(root->left, min,root->val)&&isV(root->right,root->val,max)){
            
            return true;
            
            
        }else{
            return false;
        }
        
        
    }
};
