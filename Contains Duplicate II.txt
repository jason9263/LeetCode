class Solution {
    
    public HashMap<Integer,Integer> map = new HashMap<Integer,Integer>();
    
    public boolean containsNearbyDuplicate(int[] nums, int k) {
        
        for(int i = 0; i < nums.length; i++){
            if(map.containsKey(nums[i])){
                
                if((i - map.get(nums[i])) <= k){
                    return true;
                }else{
                    map.put(nums[i], i);
                }
                
            }else{
                
                map.put(nums[i],i);
                
            }
        }
        
        return false;
    }
}
