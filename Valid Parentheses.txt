public class Solution {
    public boolean isValid(String s) {
        
        if(s == null || s.length() == 0){
            return true;
        }
        
        Stack<Character> js = new Stack<>();
        
        for(int i = 0; i < s.length(); i++){
            char tmp = s.charAt(i);
            if(tmp == '[' || tmp == '(' || tmp == '{'){
                js.push(tmp);
            }else if(tmp == ']'){
                if(!js.isEmpty() &&  js.peek() == '['){
                    js.pop();
                }else{
                    return false;
                }
            }else if(tmp == '}'){
                if(!js.isEmpty() && js.peek() == '{'){
                    js.pop();
                }else{
                    return false;
                }
            }else if(tmp == ')'){
                if(!js.isEmpty() && js.peek() == '('){
                    js.pop();
                }else{
                    return false;
                }
            }
            
        }
        
        return js.isEmpty();
    }
}
