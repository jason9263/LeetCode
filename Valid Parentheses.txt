public class Solution {
    public boolean isValid(String s) {
        if(s == null || s.length() == 0){
            return true;
        }
        
        if(s.length()% 2 == 1){
            return false;
        }else{
            int len = s.length();
            Stack<Character> js = new Stack<Character>();
            
            for(int i = 0; i < len; i++){
                char tmp = s.charAt(i);
                if(tmp == '(' || tmp == '{' || tmp == '['){
                    js.push(tmp);
                }else if(tmp == ')'){
                    if(js.size() >0 && js.peek() == '('){
                        js.pop();
                    }else{
                        return false;
                    }
                }else if(tmp == '}'){
                    if(js.size() > 0 && js.peek() == '{'){
                        js.pop();
                    }else{
                        return false;
                    }
                }else if(tmp == ']'){
                    if(js.size() > 0 && js.peek() == '['){
                        js.pop();
                    }else{
                        return false;
                    }
                }
                
            }
            
            return js.size() == 0 ? true:false;
            
        }
        
        
    }
}
