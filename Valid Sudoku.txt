class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        if(board.size() == 0){
            return false;
        }
        
        
        int row[9];
        int col[9];
        

     for(int i =0; i<9; i++)  
      {  
        memset(row, 0, 9*sizeof(int));  
        memset(col, 0, 9*sizeof(int));  
        for(int j =0; j<9; j++)  
        {  
          if(board[i][j] != '.')  
          {   
           if(row[board[i][j]-49] ==1)  
             return false;  
            row[board[i][j]-49]++;  
         }  
          if(board[j][i] != '.')  
          {    
            if(col[board[j][i]-49] ==1)  
              return false;  
            col[board[j][i]-49]++;  
          }  
        }  
      } 
        
        
        
        for(int i = 0 ; i < 9; i = i+3){
            for(int j = 0 ; j < 9; j = j+3){
                
                memset(row,0,9*sizeof(int));
                
                for(int m = 0; m < 3; m++){
                    for(int n = 0 ; n <3;n++){
                        if(board[i+m][j+n] == '.'){
                            continue;
                        }
                        
                        if(row[board[i+m][j+n]-49] == 1){
                            return false;
                        }
                        
                        
                        row[board[i+m][j+n] - 49]++;
                        
                    }
                }
                
                
                
            }
        }
        
        
        
        return true;

        
        
    }
};
