public class Solution {
    public int hIndex(int[] citations) {
        int h = 0;
        
        if(citations == null || citations.length == 0 ){
            return h;
        }
        
        int len = citations.length;
        
        for(int i = 0; i < len; i++){
            if(len - i < h  ){
                break;
            }
            
            h = Math.max(h, Math.min(len - i, citations[i]));
        }
        
        
        return h;
    }
}
